==============
IMPLEMENTATION PATTERNS
==============

Setup Context:
  ctx = context.WithValue(ctx, types.TenantIDContextKey, uint64(1234))
  ctx = context.WithValue(ctx, types.UserIDContextKey, "user-id")

Extract Safely:
  tenantID, ok := types.TenantIDFromContext(ctx)
  if !ok { return err }

File Download:
  content, mime, err := svc.GetKnowledgeFile(ctx, id)
  defer content.Close()

Pagination:
  page := &types.Pagination{Offset: 0, Limit: 20}
  result, err := svc.ListPagedKnowledgeByKnowledgeBaseID(ctx, kbID, page, filter)

Streaming:
  svc.KnowledgeQA(ctx, req, eventBus)

Types:
  types.Pagination: {Offset, Limit}
  types.PageResult: {Items, Total}
  types.QARequest: query + session + KB IDs + model
  types.SearchResult: chunk + score + source
  types.KnowledgeListFilter: tag/keyword/type/status/source/time filters
  types.AgentConfig: agent settings + tools + KBs
  types.AgentState: thoughts + tools + answer + references
